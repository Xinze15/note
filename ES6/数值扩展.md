# 数值扩展
es6写法
```
console.log('0b111110111');//2进制
console.log(0o767);//8进制
console.log(Number.isFinite(15));//true 判断该数是不是有限的
console.log(Number.isFinite(NaN));//false
console.log(Number.isNaN(0));//false  判断是不是数字
console.log(Number.isInteger(25));//判断是不是整数
console.log(Number.Max_SAFE_INTEGER，Number.MIN_SAFR_INTEGER);//最大范围和最小范围
console.log(Number.isSafeInteger(10));//判断是不是在安全范围之内
console.log(Math.trunc(4.9));//4
console.log(Math.sign(-5));//-1 正数返回1 负数返回-1 0返回0 非数字返回NaN
console.log('8',Math.cbrt(8)); 立方根
```
